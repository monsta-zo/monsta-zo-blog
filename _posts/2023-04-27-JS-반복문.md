# for 문

```js
/*
- 초기문이 존재한다면 초기문이 먼저 실행됩니다.
- 조건문이 참이라면 블록 내부 코드가 실행되고, 거짓이면 반복문이 종료됩니다. 
- 블록 내부 코드가 실행된 뒤에 증감문이 실행됩니다.
*/
for (초기문; 조건문; 증감문) {
  // statements
}
// 1부터 100까지의 합 계산
let summary = 0;
for (let i = 1; i <= 100; i++) {
  summary += i;
}
console.log(summary);
```

# while 문

```js
/*
- while문은 조건문이 참일 때 실행되는 반복문입니다.
- 블록 내부의 코드가 실행되기 전에 참 혹은 거짓을 판단합니다. */
while (조건문) {
  // 실행되는 코드 부분
}
```
